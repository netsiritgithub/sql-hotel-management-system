1)	GUEST_SUMMARY_VIEW

SELECT 
    g.GUEST_ID,
    g.FIRST_NAME || ' ' || g.LAST_NAME AS GUEST_NAME,
    g.EMAIL,
    g.PHONE,

    b.ROOM_NUMBER,
    b.CHECKIN_DATE,
    b.CHECKOUT_DATE,
    b.TOTAL_COST,

    f.RATING,
    f.FEEDBACK_COMMENT,
    f.FEEDBACK_DATE

FROM GUEST g
LEFT JOIN BOOKING b
    ON g.GUEST_ID = b.GUEST_ID
LEFT JOIN FEEDBACK f
    ON g.GUEST_ID = f.GUEST_ID;

2)	DEPARTMENT_FULL_VIEW 

SELECT 
    d.DEPARTMENT_NAME,

   d.MANAGER_NAME,
  
    s.FIRST_NAME || ' ' || s.LAST_NAME AS STAFF_NAME,
    s.EMAIL AS STAFF_EMAIL,
    s.ROLE AS STAFF_ROLE
FROM DEPARTMENT d
LEFT JOIN STAFF s
    ON d.DEPARTMENT_NAME = s.DEPARTMENT_NAME;



3)	ROOM_USAGE_VIEW

SELECT
    r.ROOM_NUMBER,
    r.ROOM_TYPE,
    b.GUEST_ID,
    b.CHECKIN_DATE,
    b.CHECKOUT_DATE,
  
    m.MAINTENANCE_DATE,
    m.MAINTENANCE_DESCRIPTION
FROM ROOM r
LEFT JOIN BOOKING b ON r.ROOM_NUMBER = b.ROOM_NUMBER
LEFT JOIN MAINTENANCE m ON r.ROOM_NUMBER = m.ROOM_NUMBER;

4)	SERVICE_REQUEST_SUMMARY_VIEW

SELECT 
    sr.SERVICE_REQUEST_ID,
    sr.REQUEST_DATE,
    sr.STATUS,

    g.GUEST_ID,
    g.FIRST_NAME || ' ' || g.LAST_NAME AS GUEST_NAME,
    g.EMAIL AS GUEST_EMAIL,

    hs.HOTEL_SERVICE_NAME,
    hs.DEPARTMENT_NAME,
    hs.HOTEL_SERVICE_COST

FROM SERVICE_REQUEST sr
JOIN GUEST g
    ON sr.GUEST_ID = g.GUEST_ID
JOIN HOTEL_SERVICE hs
    ON sr.HOTEL_SERVICE_NAME = hs.HOTEL_SERVICE_NAME;


5)	MAINTENANCE_HISTORY_VIEW

SELECT
    m.TASK_ID,
    m.ROOM_NUMBER,
    r.ROOM_TYPE,
    m.MAINTENANCE_DATE,
    m.MAINTENANCE_DESCRIPTION,
    s.STAFF_ID,
    s.FIRST_NAME || ' ' || s.LAST_NAME AS STAFF_NAME,
    s.ROLE
FROM MAINTENANCE m
JOIN ROOM r ON m.ROOM_NUMBER = r.ROOM_NUMBER
JOIN STAFF s ON m.STAFF_ID = s.STAFF_ID;
